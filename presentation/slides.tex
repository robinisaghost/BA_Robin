\begin{frame}[t]{Hypoglycemia and the Need for Prediction}
  \small
  \begin{columns}[T]
    \column{0.40\textwidth}
    \begin{itemize}
      \item 537\,M adults with diabetes globally (2021), projected 783\,M by 2045 \cite{idf2021}

      \item \textbf{Hypoglycemia} (BG $<$ 70\,mg/dL): immediate, life-threatening danger

    \end{itemize}

\begin{frame}[t]{Model 1: Long Short-Term Memory \cite{hochreiter1997lstm}}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \textbf{The Idea}
    \begin{itemize}
      \item Processes the 24 lookback steps \emph{strictly chronologically}
      \item Maintains a \textbf{cell state}: an internal memory updated step by step
      \item Three \textbf{gates} regulate information flow:
        \begin{itemize}
          \small
          \item \textbf{Forget gate} decides which old trends to discard
          \item \textbf{Input gate} controls how the new reading updates memory
          \item \textbf{Output gate} regulates what is passed to the next step
        \end{itemize}
      \item Final hidden state $h_{24}$ $\to$ linear layer $\to$ forecast at $t{+}60$\,min
    \end{itemize}

    \column{0.40\textwidth}
    % Hier wird der Block nach unten geschoben. Wert nach Belieben anpassen!
    \vspace*{1cm}

    \begin{block}{Setup in this thesis}
      \begin{itemize}
        \item 2 stacked LSTMs, hidden $=$ 256
        \item Dropout $0.159$
        \item AdamW, $\text{lr}=9.61\!\times\!10^{-4}$
        \item Per-patient z-score
        \item Loss: MSE
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

% Slide 7: PatchTST Concept

\begin{frame}[t]{Model 2: PatchTST \cite{nie2023patchtst}}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \textbf{The Idea}
    \begin{itemize}
      \item Abandons step-by-step processing
      \item \textbf{Patching}: divides the 24-step window into 3 overlapping segments (length 12, stride 6)
      \item Each patch acts like a "word token", similar to NLP \cite{vaswani2017attention}
      \item \textbf{Self-attention}: every patch attends to every other patch in parallel
      \item Captures \emph{global} context at once, not sequentially
    \end{itemize}

    \column{0.40\textwidth}
    \begin{block}{Setup in this thesis}
      \begin{itemize}
        \item 3 Transformer blocks
        \item 4 attention heads, $d_{\text{model}}{=}64$
        \item Patches: length 12, stride 6
        \item AdamW, $\text{lr}=1.21\!\times\!10^{-4}$
        \item RevIN normalization \cite{kim2022revin}
        \item Loss: MSE
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

% Slide 12: Objective 1 Three Loss Functions
% =====================================================================
\begin{frame}[t,shrink=6]{Objective 1: Three Loss Functions Compared}
  \begin{columns}[T]
    \column{0.32\textwidth}
    \textbf{Baseline MSE}
    \[ \mathcal{L} = \mathrm{MSE}(\hat{y},\,y) \]
    \begin{itemize}
      \small
      \item Penalises the target point directly
      \item No notion of timing tolerance
      \item Reference for comparison
    \end{itemize}

    \column{0.32\textwidth}
    \textbf{1a Bounded-Lag}
    \[ \mathcal{L} = \min_{\Delta \in [-D,D]}\,\mathrm{MSE}(\hat{y},\,y_{\text{shift}(\Delta)}) \]
    \begin{itemize}
      \small
      \item $D{=}3$ ($\pm 15$\,min)
      \item Picks the best shift per sample
    \end{itemize}

    \column{0.32\textwidth}
    \textbf{1b Soft-DTW \cite{cuturi2017sdtw}}
    \[ \mathcal{L} = \mathrm{softDTW}_{\gamma}(\hat{y},\,y) \]
    \begin{itemize}
      \small
      \item Differentiable elastic alignment
      \item Allows non-uniform warping
      \item $\gamma{=}1$ (smooth-min sharpness)
    \end{itemize}
  \end{columns}
  \vspace{2pt}
  \begin{alertblock}{Hypothesis}
    If MSE punishes the model for offsets too harshly, a forgiving loss should reduce the systematic shift.
  \end{alertblock}
\end{frame}

% =====================================================================
% Slide 13 Soft DTW (Optimiertes 2-Spalten Layout mit dynamischen Citations)
% =====================================================================
\begin{frame}[t]{Soft DTW Differentiable Sequence Alignment}
  \vspace{-15pt}

  \begin{columns}[T]

    % Linke Spalte (75% der Breite) für die Formeln
    \begin{column}{0.75\textwidth}

      \textbf{Classical DTW Loss \cite{sakoe1978dtw}} \\[4pt]
      \footnotesize
      The goal is to find an optimal alignment path $A$ that minimizes the mapping cost
      \[
        \mathcal{L}_{DTW}(x, y) = \min_{A \in \mathcal{A}} \langle A, \Delta(x, y) \rangle
      \]
      This discrete minimum is not differentiable and prevents backpropagation

      \vspace{15pt}

      \normalsize\textbf{Soft Min Operator} \\[4pt]
      \footnotesize
      Differentiability is achieved through this smooth approximation
      \[
        \min\nolimits_{\gamma}^\ast \{a_1, \dots, a_k\} = -\gamma \log \sum_{i=1}^k e^{-a_i/\gamma}
      \]

      \vspace{-5pt}

      \normalsize\textbf{Soft DTW Loss \cite{cuturi2017sdtw}} \\[4pt]
      \footnotesize
      The loss is smoothed by replacing the minimum with a log sum exp operator
      \[
        \mathcal{L}_{\gamma}(x, y) = -\gamma \log \sum_{A \in \mathcal{A}} e^{-\langle A, \Delta(x, y) \rangle / \gamma}
      \]

    \end{column}

  \end{columns}
\end{frame}

% Slide 14: Soft-DTW Mechanics in Detail (Numerical Example)

\begin{frame}[t]{Soft-DTW Calculation: Step-by-Step}
  \centering
  \textbf{Input Sequences ($H{=}3$)} \\
  Prediction $\hat{y} = [10, 10, 6]$ \quad vs. \quad Ground Truth $y = [10, 8, 6]$
  \vspace{0.5em}

  \begin{columns}[T]
    \column{0.53\textwidth}
    \centering
    \textbf{Distance Matrix $D$} \\
    $D_{i,j} = (\hat{y}_i - y_j)^2$
    \[
    D = \begin{pmatrix}
    \mathbf{0} & 4 & 16 \\
    0 & 4 & 16 \\
    16 & 4 & \mathbf{0}
    \end{pmatrix}
    \]

    \column{0.48\textwidth}
    \centering
    \textbf{DP-Table $R$} \\
    $R_{i,j} = D_{i,j} + \mathrm{softmin}_{\gamma}(d,t,l)$
    \[
    R = \begin{pmatrix}
    \mathbf{0} & \infty & \infty & \infty \\
    \infty & 0 & 4 & 20 \\
    \infty & 0 & 3.3 & 19 \\
    \infty & 16 & 19.3 & \mathbf{3.3}
    \end{pmatrix}
    \]
  \end{columns}

  \vspace{-0.5em}

  \begin{block}{Mathematical Operation (for $\gamma{=}1$)}
    \centering
    $\mathrm{softmin}(d,t,l) = -\ln(e^{-d} + e^{-t} + e^{-l})$ \\[0.3em]
    \textbf{Final Loss $R_{3,3} \approx 3.3$}
  \end{block}
\end{frame}

% Slide 16: Objective 2 Multi-step concept
% =====================================================================
\begin{frame}[t]{Objective 2: Joint Multi-Step MSE (MIMO)}
  \textbf{Idea:} supervise all 12 future steps, not just $h{=}11$.
  \vspace{8pt}
  \begin{columns}[T]
    \column{0.45\textwidth}
    \textbf{Training objective}
    \[ \mathcal{L} = \frac{1}{12}\sum_{h=0}^{11}\mathrm{MSE}(\hat{y}_h,\,y_h) \]
    \begin{itemize}
      \item Output head: $(B, 1) \to (B, 12)$
      \item Same architecture, same data
      \item All other settings identical to baseline
    \end{itemize}

    \column{0.50\textwidth}
    \textbf{Why this might help}\\[4pt]
    Supervising early steps ($h{=}1, 2, 3, \ldots$) forces the model to commit to a \emph{trajectory shape}, not just a 60-min endpoint. \\[4pt]
    If a 5-min prediction is already late, the gradient signal corrects it \emph{immediately}, before the lag can compound over the horizon.
  \end{columns}
\end{frame}

\begin{frame}[t]{Objective 3b: Direct Binary Classifier}
  \textbf{Task:} $p(\text{hypoglycemia within 60 min}) \in [0, 1]$
  \vspace{6pt}
  \begin{columns}[T]
    \column{0.50\textwidth}
    \textbf{Architecture}
    \begin{itemize}
      \item Same LSTM / PatchTST backbone as before
      \item Same hyperparameters as baseline
      \item Output: single sigmoid scalar
      \item Label: $1$ if $\min(y_{t+1:t+12}) < 70$\,mg/dL
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Loss: Binary Cross-Entropy with Logits}
    \[ \mathcal{L} = -w_{+} \cdot y\,\log\sigma(z) - (1{-}y)\log(1{-}\sigma(z)) \]
    \begin{itemize}
      \small
      \item $w_{+} = n_{\text{neg}}/n_{\text{pos}}$ counters class imbalance (hypos are rare)
      \item Decision threshold $0.5$
    \end{itemize}
  \end{columns}
\end{frame}

    % Rechte Spalte: Kompakte Mechanismenbeschreibung
    \begin{column}{0.26\textwidth}
      \textbf{The mechanism}
      \vspace{4pt}
      \begin{itemize}
        \scriptsize
        \item The classification head on top of PatchTST produces very large logits
        \item These values are passed through the sigmoid activation and probabilities collapse to $\approx 0$ or $\approx 1$
        \item The standard decision threshold of 0.5 becomes irrelevant
        \item LSTM produces smoothly varying probabilities because the recurrent state bounds the internal values
      \end{itemize}

    \end{column}

\begin{frame}[t]{Conclusions}
  \vspace{-10pt}

  \begin{enumerate}
    \setlength{\itemsep}{18pt} % Etwas mehr Platz, da es jetzt nur noch 3 Punkte sind

    \item \textbf{Time-Shift is a Fundamental Limit} \\[2pt]
    { Without future causal data (e.g., meals, insulin), models inherently propagate past values. Alternative loss functions cannot resolve this structural delay.\par}

    \item \textbf{Classification Outperforms Regression for Alerts} \\[2pt]
    { Pointwise regression fails at predicting isolated threshold crossings. Reliable hypoglycemia event detection requires task-specific direct binary classification.\par}

    \item \textbf{Complex Architectures Require Calibration} \\[2pt]
    { Modern models like PatchTST exhibit severe overconfidence in this domain. Targeted calibration is essential for safe clinical application.\par}

  \end{enumerate}
\end{frame}

\begin{frame}[t,shrink=7]{Future Work}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Short-term:}
    \begin{itemize}
      \item Threshold optimization for the LSTM classifier (we used the default $0.5$)
      \item \textbf{Calibration} for PatchTST: temperature scaling \cite{guo2017calibration}
      \item \textbf{Slope-aware losses} that explicitly penalise the wrong direction or magnitude of glucose change. The DILATE loss \cite{leguen2019dilate} combines a shape term and a temporal term. It might force the model to first match \emph{how fast} glucose is changing.
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Long-term:}
    \begin{itemize}
      \item \textbf{Multi-modal input} (insulin records, announced meals)
      \item Population-level pretraining with patient-specific fine-tuning
    \end{itemize}
  \end{columns}
